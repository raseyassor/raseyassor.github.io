import pandas as pd
import numpy as np

# Calcul rapide d'autoconsommation PV
def self_consumption(csv_load, pv):
    df = pd.read_csv(csv_load)
    sc = np.minimum(df['load'], pv)
    ratio = 100 * sc.sum() / pv.sum()
    return f"Autoconsommation: {ratio:.1f}%"

# Output: Autoconsommation: 82.7%
